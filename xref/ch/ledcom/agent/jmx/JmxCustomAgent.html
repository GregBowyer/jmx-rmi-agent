16  package ch.ledcom.agent.jmx;
17  
18  import javax.management.MBeanServer;
19  import javax.management.remote.JMXConnectorServer;
20  import javax.management.remote.JMXConnectorServerFactory;
21  import javax.management.remote.JMXServiceURL;
22  import java.io.IOException;
23  import java.lang.management.ManagementFactory;
24  import java.net.InetAddress;
25  import java.rmi.registry.LocateRegistry;
26  import java.util.HashMap;
27  
28  /**
29   * This CustomAgent will start an RMI Connector Server using only port
30   * "ch.ledcom.agent.jmx.port". When used with the javaagent option, it allows a
31   * remote connection using JConsole or VisualVM through an SSH tunnel.
32   * <p/>
33   * The purpose of this agent is to open a JMX server that will use only one
34   * fixed port for all communication instead of two dynamically set ports.
35   * <p/>
36   * Code adapted from:
37   * https://blogs.oracle.com/jmxetc/entry/connecting_through_firewall_using_jmx
38   *
39   * @author gehel
40   */
41  public final class JmxCustomAgent {
42  
43      /**
44       * Base name for all parameter names.
45       */
46      private static final String BASE = "ch.ledcom.agent.jmx.";
47  
48      /** Parameter name to set the JMX port. */
49      public static final String PORT_KEY = BASE + "port";
50  
51      /** Parameter name to force localhost. */
52      public static final String FORCE_LOCALHOST_KEY = BASE + "forceLocalhost";
53  
54      /** Parameter name to activate authentication. */
55      public static final String AUTHENTICATE = BASE + "authenticate";
56  
57      /** Parameter name to set password file used for authentication. */
58      public static final String PASSWORD_FILE = BASE + "password.file";
59  
60      /** Parameter name to set access file used for authentication. */
61      public static final String ACCESS_FILE = BASE + "access.file";
62  
63      /** Default JMX port. */
64      private static final int DEFAULT_PORT = 3000;
65  
66      /** Private constructor to make sure nobody tries to instantiate this class. */
67      private JmxCustomAgent() {
68      }
69  
70      /**
71       * Entry point into a JavaAgent.
72       *
73       * @param agentArgs arguments given to the agent (not used, we only use
74       *                  system properties)
75       * @throws IOException in case of problems setting up the agent
76       */
77      public static void premain(final String agentArgs) throws IOException {
78          // get all parameters from environment
79          final int port = parseInt(System.getProperty(PORT_KEY), DEFAULT_PORT);
80          final boolean forceLocalhost = Boolean.getBoolean(FORCE_LOCALHOST_KEY);
81          final String accessFile = System.getProperty(ACCESS_FILE);
82          final String passwordFile = System.getProperty(PASSWORD_FILE);
83          final boolean authenticate = Boolean.getBoolean(AUTHENTICATE);
84  
85          System.out.println("RMI / JMX Custom Agent");
86          System.out.println("======================");
87          System.out.println(PORT_KEY + " = [" + port + "]");
88          System.out.println(FORCE_LOCALHOST_KEY + " = [" + forceLocalhost + "]");
89          System.out.println(AUTHENTICATE + " = [" + authenticate + "]");
90          System.out.println(ACCESS_FILE + " = [" + accessFile + "]");
91          System.out.println(PASSWORD_FILE + " = [" + passwordFile + "]");
92  
93          // Ensure cryptographically strong random number generator used
94          // to choose the object number - see java.rmi.server.ObjID
95          System.setProperty("java.rmi.server.randomIDs", "true");
96  
97          System.out.println("Create RMI registry on port " + port);
98          LocateRegistry.createRegistry(port);
99  
100         // Retrieve the PlatformMBeanServer.
101         MBeanServer mbs = ManagementFactory.getPlatformMBeanServer();
102 
103         // Environment map.
104         System.out.println("Initialize the environment map");
105         HashMap<String, Object> env = new HashMap<String, Object>();
106 
107         // Provide the password file used by the connector server to
108         // perform user authentication. The password file is a properties
109         // based text file specifying username/password pairs.
110         env.put("jmx.remote.x.password.file", passwordFile);
111 
112         // Provide the access level file used by the connector server to
113         // perform user authorization. The access level file is a properties
114         // based text file specifying username/access level pairs where
115         // access level is either "readonly" or "readwrite" access to the
116         // MBeanServer operations.
117         env.put("jmx.remote.x.access.file", accessFile);
118         env.put("jmx.remote.x.authenticate", authenticate);
119 
120         // Create an RMI connector server.
121         //
122         // As specified in the JMXServiceURL the RMIServer stub will be
123         // registered in the RMI registry running in the local host on
124         // port 3000 with the name "jmxrmi". This is the same name the
125         // out-of-the-box management agent uses to register the RMIServer
126         // stub too.
127         //
128         // The port specified in "service:jmx:rmi://"+hostname+":"+port
129         // is the second port, where RMI connection objects will be exported.
130         // Here we use the same port as that we choose for the RMI registry.
131         // The port for the RMI registry is specified in the second part
132         // of the URL, in "rmi://"+hostname+":"+port
133         String hostname;
134         if (forceLocalhost) {
135             System.out
136                     .println("Using localhost as hostname "
137                             + "as defined by system property: "
138                             + FORCE_LOCALHOST_KEY);
139             hostname = "localhost";
140         } else {
141             hostname = InetAddress.getLocalHost().getHostName();
142         }
143 
144         System.out.println("Create an RMI/JMX connector server, hostname is: ["
145                 + hostname + "]");
146         String serviceUrl = "service:jmx:rmi://" + hostname + ":" + port
147                 + "/jndi/rmi://" + hostname + ":" + port + "/jmxrmi";
148         System.out.println("and url is: [" + serviceUrl + "]");
149         JMXServiceURL url = new JMXServiceURL(serviceUrl);
150 
151         // Now create the server from the JMXServiceURL
152         JMXConnectorServer cs = JMXConnectorServerFactory
153                 .newJMXConnectorServer(url, env, mbs);
154 
155         // Start the RMI connector server.
156         System.out.println("Start the RMI/JMX connector server on port ["
157                 + port + "]");
158         cs.start();
159     }
160 
161     /**
162      * Utility method to parse integer.
163      *
164      * @param s            String to parse as an int
165      * @param defaultValue default value to return in case the String is empty
166      * @return the integer represented by the String or the default value if
167      * the String is empty
168      */
169     private static int parseInt(final String s, final int defaultValue) {
170         if (s == null || s.trim().equals("")) {
171             return defaultValue;
172         }
173         return Integer.parseInt(s);
174     }
175 }
